## Implementation

### cli/cache.ts

```typescript
import { existsSync, readFileSync, writeFileSync, mkdirSync } from 'fs';
import { join, dirname } from 'path';
import { $ } from 'bun';

const CACHE_DIR = '.agents/qa-scan/cache';

export function getCachePath(repoKey: string): string {
  return join(process.cwd(), CACHE_DIR, repoKey);
}

export function isCacheValid(repo: RepoConfig): boolean {
  const cachePath = getCachePath(repo.key);
  const commitFile = join(cachePath, '.analyzed-commit');
  
  if (!existsSync(commitFile)) return false;
  
  const cachedCommit = readFileSync(commitFile, 'utf-8').trim();
  const currentCommit = getGitHead(repo.path);
  
  return cachedCommit === currentCommit;
}

export function loadRoadmap(repo: RepoConfig): TestRoadmap | null {
  const roadmapPath = join(getCachePath(repo.key), 'test-roadmap.json');
  
  if (!existsSync(roadmapPath)) return null;
  
  return JSON.parse(readFileSync(roadmapPath, 'utf-8'));
}

export function writeRoadmap(repo: RepoConfig, roadmap: TestRoadmap): void {
  const cachePath = getCachePath(repo.key);
  mkdirSync(cachePath, { recursive: true });
  
  writeFileSync(join(cachePath, 'test-roadmap.json'), JSON.stringify(roadmap, null, 2));
  writeFileSync(join(cachePath, '.analyzed-commit'), roadmap.commit);
  
  // Generate setup script if needed
  if (roadmap.test_environment.setup_commands.length > 0) {
    const script = [
      '#!/bin/bash',
      '# Auto-generated by qa-scan analyze',
      ...roadmap.test_environment.setup_commands
    ].join('\n');
    writeFileSync(join(cachePath, 'setup-script.sh'), script);
  }
}

export async function getGitHead(repoPath: string): Promise<string> {
  const result = await $`git -C ${repoPath} rev-parse HEAD`.text();
  return result.trim();
}
```
